\newenvironment{membs}{%
    \vspace{-\parskip}\par%
    \let\tmp@sep\@sep%
    \let\tmp@and\and%
    \renewcommand{\@sep}{&}%
    \begin{spacing}{1}%
}{%
    \end{spacing}%
    \let\@sep\temp@sep%
    \let\and\temp@and%
}

\newenvironment{kwords}{%
    \newcounter{kw}
    \setcounter{kw}{1}
    \let\tmp@and\and
    \renewcommand{\and}{%
        \addtocounter{kw}{1}
        \thekw. 
    }
    \thekw. 
}{%
    \let\and\temp@and
}

\newenvironment{subs}{%
    \let\tmp@sep\@sep
    \let\tmp@and\and
    \renewcommand{\@sep}{\\}
    \renewcommand{\and}{%
        \end{tabular}
        \par
        \vfill
        \begin{tabular}[t]{c}
    }
}{%
    \let\@sep\temp@sep
    \let\and\temp@and
}

\renewenvironment{thebibliography}[1]
{%
    \chapter*{\bibname
    \refstepcounter{chapter}
    \addcontentsline{toc}{chapter}{\bibname}
    \@mkboth{\MakeUppercase\bibname}{\MakeUppercase\bibname}}%      
    \singlespacing			
        \flushleft				
        \list{\@biblabel{\@arabic\c@enumiv}}%
        {\settowidth\labelwidth{\@biblabel{#1}}%
        \leftmargin\labelwidth
        \advance\leftmargin\labelsep
        \@openbib@code
        \usecounter{enumiv}%
        \let\p@enumiv\@empty
        \renewcommand\theenumiv{\@arabic\c@enumiv}}%
    \sloppy
    \clubpenalty4000
    \@clubpenalty \clubpenalty
    \widowpenalty4000%
    \sfcode`\.\@m
}{%
    \def\@noitemerr
    {\@latex@warning{Empty `thebibliography' environment}}%
    \endlist%
}
