\renewcommand\maketitle{%
	\begin{titlepage}%    
		{\noindent\large\titletext\par}
		\vfill\par
		\begin{center}
			{\bf
				{\large\@author\par}
				\vfill\par
				\Large\begin{doublespace}
					\MakeUppercase{\@title}
				\end{doublespace}
				\vfill\par
			}\par
			\aprovedtext\par
			\vfill\par
			{\large
				\sub@line
				\location\\
				\city, \uf{} - \country\\
				\date@year
			}
		\end{center}
	\end{titlepage}
}
