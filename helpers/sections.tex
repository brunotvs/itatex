\ProvidesFile{sections.tex}

\def\numberline#1{#1~}

\def\@chapter[#1]#2{\ifnum \c@secnumdepth >\m@ne
        \if@appendix
            \refstepcounter{chapter}%
            \typeout{\thechapter.}%
            \addcontentsline{toc}{chapter}%
            {\protect\numberline{\@chapapp{} \thechapter~-}#1}%
        \else
            \if@mainmatter
                \refstepcounter{chapter}%
                \typeout{\thechapter.}%
                \addcontentsline{toc}{chapter}%
                {\protect\numberline{\thechapter}#1}%
            \else
                \addcontentsline{toc}{chapter}{#1}%
            \fi
        \fi
    \else
        \addcontentsline{toc}{chapter}{#1}%
    \fi
    \chaptermark{#1}%
    \addtocontents{lof}{\protect\addvspace{10\p@}}%
    \addtocontents{lot}{\protect\addvspace{10\p@}}%
    \if@twocolumn
        \@topnewpage[\@makechapterhead{#2}]%
    \else
        \@makechapterhead{#2}%
        \@afterheading
    \fi
	\ifdefined\NR@gettitle%
		\NR@gettitle{#1}%
	\fi
}

\def\@makechapterhead#1{%
    \if@mainmatter
        {\parindent \z@ \raggedright \normalfont
        \interlinepenalty\@M
        \Large \bfseries \thechapter~~#1\par\nobreak
        \vskip 2ex\@minus\parskip
        }
    \else
        \if@appendix
            {\parindent \z@ \raggedright \normalfont
            \interlinepenalty\@M
            \Large \bfseries \@chapapp{} \thechapter\space-\space#1\par\nobreak
            \vskip 2ex\@minus\parskip
            }
        \else
            {\centering%
            \interlinepenalty\@M
            \Large \bfseries #1\par\nobreak
            \vskip 2ex\@minus\parskip
            }
        \fi
    \fi

}

\def\@makeschapterhead#1{%
    \if@mainmatter
        {\parindent \z@ \raggedright \normalfont
        \interlinepenalty\@M
        \Large \bfseries #1\par\nobreak
        \vskip 2ex\@minus\parskip
        }
    \else
        {\centering%
        \interlinepenalty\@M
        \Large \bfseries #1\par\nobreak
        \vskip 2ex\@minus\parskip
        }
    \fi
}

\renewcommand\tableofcontents{%
    \if@twocolumn
      \@restonecoltrue\onecolumn
    \else
      \@restonecolfalse
    \fi
    \chapter*{\contentsname
        \@mkboth{%
           \MakeUppercase\contentsname}{\MakeUppercase\contentsname}}%
           \vskip -1ex \@plus\parskip
    \@starttoc{toc}%
    \addtocontents{toc}{\protect\setlength{\parskip}{0pt}}%
    \if@restonecol\twocolumn\fi
}

\renewcommand\listoffigures{%
    \if@twocolumn
      \@restonecoltrue\onecolumn
    \else
      \@restonecolfalse
    \fi
    \chapter*{\listfigurename}%
      \@mkboth{\MakeUppercase\listfigurename}%
              {\MakeUppercase\listfigurename}%
    \@starttoc{lof}\addtocontents{lof}{\protect\setlength{\parskip}{0pt}}%%
    \if@restonecol\twocolumn\fi
    }

\renewcommand\listoftables{%
    \if@twocolumn
      \@restonecoltrue\onecolumn
    \else
      \@restonecolfalse
    \fi
    \chapter*{\listtablename}%
      \@mkboth{%
          \MakeUppercase\listtablename}%
         {\MakeUppercase\listtablename}%
    \@starttoc{lot}\addtocontents{lot}{\protect\setlength{\parskip}{0pt}}%%
    \if@restonecol\twocolumn\fi
}

% Redefining the \@caption command instead of using 
% patchcmd
\long\def\@caption#1[#2]#3{%
    \par
    \addcontentsline{\csname ext@#1\endcsname}{#1}%
        {\protect\numberline{\csname fnum@#1\endcsname:}{\ignorespaces #2}}%
    \begingroup
        \@parboxrestore
        \if@minipage
            \@setminipage
        \fi
        \normalsize
        \@makecaption{\csname fnum@#1\endcsname}{\ignorespaces #3}\par
  \endgroup}
